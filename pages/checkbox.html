Checkbox

사용자가 항목을 선택하거나 해제하는 체크박스 컴포넌트입니다. data-checkbox 속성을 숨겨진 input[type="checkbox"]에 지정하고, 인접한 label[data-checkbox-label]의 ::before/::after 가상 요소로 커스텀 박스와 체크마크를 렌더링합니다. data-size로 sm / md / lg 크기를 지정하며, 에러 상태, 비활성화, Indeterminate(불확정) 상태를 지원합니다.

Default Checkbox

기본 체크박스입니다. data-size 속성으로 sm(32px) / md(44px) / lg(52px) 크기를 지정합니다. 언체크·체크 두 가지 상태를 모두 확인할 수 있습니다. 

<!-- 기본 체크박스 (크기: sm / md / lg) -->
<!-- input[data-checkbox] + label[data-checkbox-label] 인접 쌍 -->

<!-- sm: 소형 (min-height 32px) -->
<input type="checkbox" data-checkbox id="cb-sm" name="cb-sm" />
<label data-checkbox-label data-size="sm" for="cb-sm">Small</label>

<!-- md: 기본(default) 크기 (min-height 44px) -->
<input type="checkbox" data-checkbox id="cb-md" name="cb-md" />
<label data-checkbox-label for="cb-md">Medium</label>

<!-- lg: 대형 (min-height 52px) -->
<input type="checkbox" data-checkbox id="cb-lg" name="cb-lg" />
<label data-checkbox-label data-size="lg" for="cb-lg">Large</label>

<!-- checked 상태: checked 속성 추가 -->
<input type="checkbox" data-checkbox id="cb-md-c" name="cb-md-c" checked />
<label data-checkbox-label for="cb-md-c">Medium (checked)</label>
/* 네이티브 체크박스 시각적 숨김 (접근성 트리 유지) */
input[type="checkbox"][data-checkbox] {
  position: absolute;
  width: 1px; height: 1px;
  padding: 0; margin: -1px;
  overflow: hidden;
  clip: rect(0, 0, 0, 0);
  white-space: nowrap; border: 0;
}

/* 커스텀 체크박스 레이블: 수평 배치 + 터치 타겟 */
label[data-checkbox-label] {
  display: inline-flex;
  align-items: center;
  gap: var(--spacing-sm);
  min-height: 44px; /* WCAG 2.5.5 */
  cursor: pointer;
  user-select: none;
  position: relative;
}

/* ::before: 체크박스 박스(외형) */
label[data-checkbox-label]::before {
  content: "";
  width: 18px; height: 18px;
  flex-shrink: 0;
  box-sizing: border-box;
  border: 1px solid var(--color-border-hover);
  border-radius: 3px;
  background-color: var(--color-bg);
}

/* 체크 상태 */
input[type="checkbox"][data-checkbox]:checked + label[data-checkbox-label]::before {
  background-color: var(--color-primary);
  border-color: var(--color-primary);
}
input[type="checkbox"][data-checkbox]:checked + label[data-checkbox-label]::after {
  opacity: 1;
}

/* 크기 변형 */
label[data-checkbox-label][data-size="sm"] { font-size: var(--font-size-sm); min-height: 32px; }
label[data-checkbox-label][data-size="sm"]::before { width: 14px; height: 14px; }
label[data-checkbox-label][data-size="lg"] { font-size: var(--font-size-lg); min-height: 52px; }
label[data-checkbox-label][data-size="lg"]::before { width: 22px; height: 22px; border-radius: 4px; }
// 체크 상태 변경 감지 예시
const checkbox = document.querySelector('[data-checkbox]');
checkbox.addEventListener('change', (e) => {
  console.log('체크 상태:', e.target.checked);
});

With Label & Helper

aria-describedby로 헬퍼 텍스트를 스크린리더에 전달합니다. 관련 항목을 그룹화할 때는 fieldset + legend를 사용합니다.

서비스 이용을 위해 약관 동의가 필요합니다.

관심 분야 *

하나 이상 선택해 주세요.

알림 설정

<!-- 단일 체크박스: 헬퍼 텍스트 + aria-describedby -->
<input
  type="checkbox" data-checkbox
  id="terms" name="terms"
  aria-describedby="terms-helper"
/>
<label data-checkbox-label for="terms">
  이용약관에 동의합니다
  <span class="checkbox-required" aria-hidden="true">*</span>
</label>
<p class="checkbox-helper" id="terms-helper">
  서비스 이용을 위해 약관 동의가 필요합니다.
</p>

<!-- 그룹 체크박스: fieldset + legend (접근성 필수) -->
<fieldset style="border: none; padding: 0;">
  <legend class="checkbox-group-label">관심 분야</legend>
  <div class="checkbox-group" aria-describedby="interest-helper">
    <div>
      <input type="checkbox" data-checkbox id="dev" name="interest" value="dev" />
      <label data-checkbox-label for="dev">개발</label>
    </div>
    <div>
      <input type="checkbox" data-checkbox id="design" name="interest" value="design" checked />
      <label data-checkbox-label for="design">디자인</label>
    </div>
  </div>
  <p class="checkbox-helper" id="interest-helper">하나 이상 선택해 주세요.</p>
</fieldset>
/* 수직 체크박스 그룹 */
.checkbox-group {
  display: flex;
  flex-direction: column;
  gap: var(--spacing-sm);
}

/* 수평 체크박스 그룹 */
.checkbox-group--inline {
  flex-direction: row;
  flex-wrap: wrap;
  gap: var(--spacing-md);
  align-items: center;
}

/* 그룹 레이블 */
.checkbox-group-label {
  display: block;
  font-size: var(--font-size-sm);
  font-weight: 500;
  color: var(--color-text);
  margin-bottom: var(--spacing-xs);
}

/* 헬퍼 텍스트 */
.checkbox-helper {
  font-size: var(--font-size-sm);
  color: var(--color-text-secondary);
  margin: 0;
  margin-top: var(--spacing-xs);
}
// 그룹 체크박스 선택 값 수집 예시
const interestBoxes = document.querySelectorAll('[name="interest"]');
interestBoxes.forEach((cb) => {
  cb.addEventListener('change', () => {
    const selected = [...interestBoxes]
      .filter((c) => c.checked)
      .map((c) => c.value);
    console.log('선택된 관심 분야:', selected);
  });
});

Error State

필수 항목 미선택 등 유효성 검사 실패 시 data-error="true"와 aria-invalid="true", 에러 메시지를 함께 제공합니다.

필수 약관에 동의해 주세요.

알림 수신 방법 *

알림 수신 방법을 하나 이상 선택해 주세요. 

<!-- 에러 상태: data-error + aria-invalid + role="alert" -->
<input
  type="checkbox" data-checkbox
  data-error="true"
  id="error-terms" name="terms"
  aria-invalid="true"
  aria-describedby="error-terms-msg"
/>
<label data-checkbox-label for="error-terms">
  이용약관에 동의합니다
  <span class="checkbox-required" aria-hidden="true">*</span>
</label>
<p class="checkbox-error-msg" id="error-terms-msg" role="alert">
  <i data-lucide="alert-circle" aria-hidden="true"></i>
  필수 약관에 동의해 주세요.
</p>

<!-- 그룹 에러: fieldset 내 모든 체크박스에 data-error 적용 -->
<fieldset style="border: none; padding: 0;">
  <legend class="checkbox-group-label">알림 수신 방법 *</legend>
  <div class="checkbox-group" aria-describedby="error-notify-msg" aria-invalid="true">
    <div>
      <input type="checkbox" data-checkbox data-error="true"
        id="err-email" name="notify" aria-invalid="true" />
      <label data-checkbox-label for="err-email">이메일</label>
    </div>
  </div>
  <p class="checkbox-error-msg" id="error-notify-msg" role="alert">
    <i data-lucide="alert-circle" aria-hidden="true"></i>
    하나 이상 선택해 주세요.
  </p>
</fieldset>
/* 에러 상태: ::before 박스에 빨간 테두리 적용 */
input[type="checkbox"][data-checkbox][data-error="true"]
  + label[data-checkbox-label]::before {
  border-color: var(--color-danger);
  background-color: color-mix(in srgb, var(--color-danger) 6%, var(--color-bg));
}

/* 에러 + 체크 상태: 빨간 배경 */
input[type="checkbox"][data-checkbox][data-error="true"]:checked
  + label[data-checkbox-label]::before {
  background-color: var(--color-danger);
  border-color: var(--color-danger);
}

/* 에러 메시지 */
.checkbox-error-msg {
  display: flex;
  align-items: center;
  gap: 4px;
  font-size: var(--font-size-sm);
  color: var(--color-danger);
  margin: 0;
  margin-top: var(--spacing-xs);
}
// 폼 제출 시 필수 체크박스 유효성 검사
const form = document.querySelector('form');
form.addEventListener('submit', (e) => {
  const termsBox = document.querySelector('#error-terms');
  const errorMsg = document.querySelector('#error-terms-msg');

  if (!termsBox.checked) {
    e.preventDefault();
    termsBox.dataset.error = 'true';
    termsBox.setAttribute('aria-invalid', 'true');
    errorMsg.hidden = false;
    termsBox.focus();
  } else {
    delete termsBox.dataset.error;
    termsBox.setAttribute('aria-invalid', 'false');
    errorMsg.hidden = true;
  }
});

Disabled Checkbox

disabled 속성 추가 시 반투명 처리되고 클릭이 차단됩니다. 스크린리더가 비활성 상태를 자동으로 안내합니다. 비활성 + 체크 상태는 현재 선택된 읽기 전용 값을 표시할 때 사용합니다. 

<!-- 비활성 + 미체크 -->
<input
  type="checkbox" data-checkbox
  id="disabled-unchecked"
  name="disabled-unchecked"
  disabled
/>
<label data-checkbox-label for="disabled-unchecked">
  Disabled
</label>

<!-- 비활성 + 체크: 읽기 전용 선택 상태 -->
<input
  type="checkbox" data-checkbox
  id="disabled-checked"
  name="disabled-checked"
  disabled
  checked
/>
<label data-checkbox-label for="disabled-checked">
  Disabled (checked)
</label>
/* 비활성 상태: 반투명 + 커서 변경 + 클릭 차단 */
input[type="checkbox"][data-checkbox]:disabled + label[data-checkbox-label] {
  opacity: 0.5;
  cursor: not-allowed;
  pointer-events: none;
}

/* 비활성 + 체크 상태: primary 배경 유지 (상태 인지 보장) */
input[type="checkbox"][data-checkbox]:disabled:checked + label[data-checkbox-label]::before {
  background-color: var(--color-primary);
  border-color: var(--color-primary);
}
// 조건부 비활성화 예시
const checkboxEl = document.querySelector('#disabled-unchecked');
const toggleBtn  = document.querySelector('#enable-btn');

toggleBtn.addEventListener('click', () => {
  checkboxEl.disabled = !checkboxEl.disabled;
  toggleBtn.textContent = checkboxEl.disabled ? '활성화' : '비활성화';
});

Indeterminate State

일부 하위 항목만 선택된 부모 체크박스에 사용하는 불확정(tri-state) 상태입니다. aria-checked="mixed" 속성으로 스크린리더에 부분 선택 상태를 전달합니다. CSS ::after를 수평 대시(–)로 재정의하여 시각적으로 표시합니다.

크기별 Indeterminate

"전체 선택" 패턴

<!-- Indeterminate: aria-checked="mixed" 속성 사용 -->
<input
  type="checkbox" data-checkbox
  id="parent-cb" name="parent-cb"
  aria-checked="mixed"
  aria-controls="child-1 child-2 child-3"
/>
<label data-checkbox-label for="parent-cb">
  전체 선택
</label>

<!-- 자식 체크박스들 -->
<div class="checkbox-group" style="margin-left: 26px;">
  <div>
    <input type="checkbox" data-checkbox id="child-1" name="items" value="1" checked />
    <label data-checkbox-label for="child-1">항목 1</label>
  </div>
  <div>
    <input type="checkbox" data-checkbox id="child-2" name="items" value="2" />
    <label data-checkbox-label for="child-2">항목 2</label>
  </div>
  <div>
    <input type="checkbox" data-checkbox id="child-3" name="items" value="3" checked />
    <label data-checkbox-label for="child-3">항목 3</label>
  </div>
</div>
/* Indeterminate: ::before 박스를 primary 배경으로 */
input[type="checkbox"][data-checkbox][aria-checked="mixed"]
  + label[data-checkbox-label]::before {
  background-color: var(--color-primary);
  border-color: var(--color-primary);
}

/* Indeterminate: ::after를 수평 대시(-)로 재정의 */
input[type="checkbox"][data-checkbox][aria-checked="mixed"]
  + label[data-checkbox-label]::after {
  width: 10px;
  height: 2px;
  border: none;
  background-color: #ffffff;
  border-radius: 1px;
  transform: translateX(4px) translateY(-50%);
  opacity: 1;
}
// "전체 선택" 체크박스 Indeterminate 관리
const parentCb = document.querySelector('#parent-cb');
const childCbs = document.querySelectorAll('[name="items"]');

function updateParentState() {
  const total   = childCbs.length;
  const checked = [...childCbs].filter((c) => c.checked).length;

  if (checked === 0) {
    parentCb.indeterminate = false;
    parentCb.checked = false;
    parentCb.setAttribute('aria-checked', 'false');
  } else if (checked === total) {
    parentCb.indeterminate = false;
    parentCb.checked = true;
    parentCb.setAttribute('aria-checked', 'true');
  } else {
    parentCb.indeterminate = true; /* 네이티브 불확정 프로퍼티 */
    parentCb.setAttribute('aria-checked', 'mixed');
  }
}

childCbs.forEach((cb) => cb.addEventListener('change', updateParentState));

parentCb.addEventListener('change', () => {
  const newState = parentCb.checked;
  childCbs.forEach((cb) => { cb.checked = newState; });
  parentCb.setAttribute('aria-checked', String(newState));
  parentCb.indeterminate = false;
});

updateParentState();

Button

사용자 액션을 트리거하는 기본 인터랙티브 요소입니다. data-variant로 5가지 스타일 변형, data-size로 sm / md / lg 크기를 지정합니다. 아이콘, 로딩 상태, 비활성화를 지원합니다.

Primary

주요 액션(저장, 제출, 확인 등)에 사용하는 기본 버튼입니다. data-variant="primary"로 지정합니다. 

<button data-variant="primary" data-size="sm" type="button">Small</button>
<button data-variant="primary" data-size="md" type="button">Medium</button>
<button data-variant="primary" data-size="lg" type="button">Large</button>
/* Primary 버튼: 파란색 배경, 흰 텍스트 */
button[data-variant="primary"] {
  background-color: var(--color-primary); /* #3b82f6 */
  border-color: var(--color-primary);
  color: #ffffff;
}

/* 크기 변형 */
button[data-size="sm"] { padding: 8px 16px; min-height: 32px; }
button[data-size="md"] { padding: 12px 20px; min-height: 40px; }
button[data-size="lg"] { padding: 16px 24px; min-height: 48px; }
/* 버튼 클릭 이벤트 예시 */
document.querySelector('button[data-variant="primary"]')
  .addEventListener('click', () => {
    console.log('Primary 버튼 클릭');
  });

Secondary

보조 액션(취소, 뒤로 가기 등)에 사용하는 회색 버튼입니다. data-variant="secondary"로 지정합니다. 

<button data-variant="secondary" data-size="sm" type="button">Small</button>
<button data-variant="secondary" data-size="md" type="button">Medium</button>
<button data-variant="secondary" data-size="lg" type="button">Large</button>
/* Secondary 버튼: 회색 배경, 흰 텍스트 */
button[data-variant="secondary"] {
  background-color: #6b7280;
  border-color: #6b7280;
  color: #ffffff;
}

/* 다크 모드: 명도 보강 (대비비 4.5:1 확보) */
:root[data-theme="dark"] button[data-variant="secondary"] {
  background-color: #9ca3af;
  border-color: #9ca3af;
  color: #111827;
}
/* Secondary 버튼 예시: 폼 취소 처리 */
document.querySelector('button[data-variant="secondary"]')
  .addEventListener('click', () => {
    history.back();
  });

Outline

투명 배경에 테두리만 있는 버튼입니다. 호버 시 primary 색상으로 전환됩니다. data-variant="outline"으로 지정합니다. 

<button data-variant="outline" data-size="sm" type="button">Small</button>
<button data-variant="outline" data-size="md" type="button">Medium</button>
<button data-variant="outline" data-size="lg" type="button">Large</button>
/* Outline 버튼: 투명 배경 + 테두리 */
button[data-variant="outline"] {
  background-color: transparent;
  border-color: var(--color-border-hover);
  color: var(--color-text);
}

/* 호버: primary 색상 테두리·텍스트로 전환 */
button[data-variant="outline"]:hover:not(:disabled) {
  background-color: var(--color-bg-secondary);
  border-color: var(--color-primary);
  color: var(--color-primary);
}
/* Outline 버튼 예시: 모달 취소 */
document.querySelector('button[data-variant="outline"]')
  .addEventListener('click', () => {
    modal.close();
  });

Ghost

배경과 테두리가 없는 버튼입니다. 호버 시에만 배경이 나타나 시각적으로 조용한 액션에 적합합니다. data-variant="ghost"로 지정합니다. 

<button data-variant="ghost" data-size="sm" type="button">Small</button>
<button data-variant="ghost" data-size="md" type="button">Medium</button>
<button data-variant="ghost" data-size="lg" type="button">Large</button>
/* Ghost 버튼: 배경·테두리 없음 */
button[data-variant="ghost"] {
  background-color: transparent;
  border-color: transparent;
  color: var(--color-text);
}

/* 호버: 배경 표시 */
button[data-variant="ghost"]:hover:not(:disabled) {
  background-color: var(--color-bg-secondary);
  border-color: var(--color-border);
}
/* Ghost 버튼 예시: 툴바 액션 */
document.querySelector('button[data-variant="ghost"]')
  .addEventListener('click', () => {
    panel.toggle();
  });

Danger

삭제·초기화 등 파괴적 액션에 사용하는 경고 버튼입니다. 사용자에게 주의를 환기합니다. data-variant="danger"로 지정합니다. 

<button data-variant="danger" data-size="sm" type="button">Small</button>
<button data-variant="danger" data-size="md" type="button">Medium</button>
<button data-variant="danger" data-size="lg" type="button">Large</button>
/* Danger 버튼: 빨간색 배경, 흰 텍스트 */
button[data-variant="danger"] {
  background-color: var(--color-danger); /* #ef4444 */
  border-color: var(--color-danger);
  color: #ffffff;
}

/* 호버: 밝기 -10% */
button[data-variant="danger"]:hover:not(:disabled) {
  filter: brightness(0.9);
}
/* Danger 버튼 예시: 삭제 확인 다이얼로그 */
document.querySelector('button[data-variant="danger"]')
  .addEventListener('click', () => {
    if (confirm('정말 삭제하시겠습니까?')) {
      deleteItem();
    }
  });

Icon

아이콘만 포함하는 정사각형 버튼과 아이콘+텍스트 조합 버튼입니다. 아이콘 전용 버튼에는 aria-label로 기능을 반드시 명시합니다.

아이콘 전용

아이콘 + 텍스트 조합

<!-- 아이콘 전용 버튼: aria-label 필수 -->
<button data-variant="primary" data-size="md" type="button" aria-label="검색">
  <i data-lucide="search" aria-hidden="true"></i>
</button>

<!-- 아이콘 + 텍스트 조합 -->
<button data-variant="primary" data-size="md" type="button">
  <i data-lucide="download" aria-hidden="true"></i>
  다운로드
</button>
/* 아이콘 전용 버튼: :has() 선택자로 정사각형 자동 적용 */
button[data-variant]:has(> i:only-child),
button[data-variant]:has(> svg:only-child) {
  padding: 0;
}

/* md 크기: 40x40px 정사각형 */
button[data-size="md"]:has(> i:only-child) {
  width: 40px;
  height: 40px;
}

/* 버튼 내부 아이콘 크기 */
button[data-variant] i { width: 1em; height: 1em; }
/* Lucide 아이콘 초기화 (common.js에서 자동 호출) */
if (window.lucide) {
  window.lucide.createIcons();
}

Loading

비동기 작업 진행 중임을 나타냅니다. data-loading="true"와 aria-busy="true"를 함께 사용합니다. 스피너는 ::after 가상 요소로 렌더링됩니다.

크기별

변형별

<!-- 로딩 버튼: data-loading + aria-busy 함께 사용 -->
<button
  data-variant="primary"
  data-size="md"
  data-loading="true"
  type="button"
  aria-busy="true"
  aria-label="처리 중"
>
  Loading...
</button>
/* 로딩 상태: cursor wait + 클릭 차단 */
button[data-variant][data-loading="true"] {
  cursor: wait;
  pointer-events: none;
}

/* 스피너: ::after 원형 테두리 회전 */
button[data-variant][data-loading="true"]::after {
  content: "";
  display: inline-block;
  width: 1em; height: 1em;
  border: 2px solid currentColor;
  border-top-color: transparent;
  border-radius: 50%;
  animation: button-spin 0.8s linear infinite;
}

@keyframes button-spin {
  from { transform: rotate(0deg); }
  to   { transform: rotate(360deg); }
}
/* 로딩 상태 전환 예시 */
async function handleSubmit(btn) {
  btn.dataset.loading = 'true';
  btn.setAttribute('aria-busy', 'true');

  try {
    await saveData();
  } finally {
    delete btn.dataset.loading;
    btn.removeAttribute('aria-busy');
  }
}

Disabled

비활성화된 버튼입니다. HTML disabled 속성을 추가하면 반투명 처리되고 클릭이 차단됩니다. 스크린리더가 비활성 상태를 자동으로 안내합니다.

크기별

변형별

<!-- Disabled: HTML disabled 속성 사용 -->
<button data-variant="primary" data-size="md" type="button" disabled>
  Disabled
</button>

<!-- 모든 변형에 동일하게 적용 -->
<button data-variant="secondary" data-size="md" type="button" disabled>Secondary</button>
<button data-variant="outline"   data-size="md" type="button" disabled>Outline</button>
<button data-variant="ghost"     data-size="md" type="button" disabled>Ghost</button>
<button data-variant="danger"    data-size="md" type="button" disabled>Danger</button>
/* Disabled 버튼: 반투명 + 커서 변경 + 클릭 차단 */
button[data-variant]:disabled,
button[data-variant][disabled] {
  opacity: 0.5;
  cursor: not-allowed;
  pointer-events: none;
  filter: none;
}
/* disabled 상태 동적 제어 예시 */
const submitBtn = document.querySelector('button[data-variant="primary"]');

function updateSubmitState(isValid) {
  submitBtn.disabled = !isValid;
}

Select

목록에서 하나의 옵션을 선택하는 드롭다운 컴포넌트입니다. 네이티브 select 요소에 data-select 속성을 지정하고, .select-wrap 컨테이너 안에 넣어 커스텀 화살표 아이콘을 배치합니다. data-size로 sm / md / lg 크기를 지정하며, data-error="true"로 에러 상태, data-success="true"로 성공 상태를 표현합니다. optgroup 요소로 옵션 그룹핑도 지원합니다.

Default Select

기본 Select 컴포넌트입니다. data-size 속성으로 sm(32px) / md(44px) / lg(52px) 크기를 지정합니다. 네이티브 appearance: none으로 기본 화살표를 제거하고 .select-icon으로 커스텀 화살표를 제공합니다.

Small (sm)

Medium (md) — 기본

Large (lg)

<!-- 기본 Select (크기: sm / md / lg) -->
<!-- .select-wrap: 커스텀 화살표 배치를 위한 컨테이너 -->

<!-- sm: 소형 -->
<div class="select-wrap">
  <select data-select data-size="sm" id="sel-sm">
    <option value="">선택하세요</option>
    <option value="kr">대한민국</option>
    <option value="us">미국</option>
  </select>
  <span class="select-icon" aria-hidden="true">
    <i data-lucide="chevron-down"></i>
  </span>
</div>

<!-- md: 기본 크기 (data-size 생략 시 md) -->
<div class="select-wrap">
  <select data-select id="sel-md">
    <option value="">선택하세요</option>
    <option value="kr" selected>대한민국</option>
    <option value="us">미국</option>
  </select>
  <span class="select-icon" aria-hidden="true">
    <i data-lucide="chevron-down"></i>
  </span>
</div>
/* 래퍼: 커스텀 화살표 절대 위치 기준 */
.select-wrap {
  position: relative;
  display: flex;
  align-items: stretch;
}

/* 네이티브 화살표 제거 + 커스텀 스타일 */
select[data-select] {
  display: block;
  width: 100%;
  box-sizing: border-box;
  padding: 10px 40px 10px 14px;
  border: 1px solid var(--color-border);
  border-radius: var(--radius-md);
  font-family: var(--font-family);
  font-size: var(--font-size-base);
  color: var(--color-text);
  background-color: var(--color-bg);
  appearance: none; /* 네이티브 화살표 제거 */
  -webkit-appearance: none;
  min-height: 44px;
  cursor: pointer;
}

/* 커스텀 화살표 */
.select-icon {
  position: absolute;
  right: 12px;
  top: 50%;
  transform: translateY(-50%);
  pointer-events: none;
  color: var(--color-text-secondary);
}
// select 값 변경 감지 예시
const sel = document.querySelector('[data-select]');
sel.addEventListener('change', (e) => {
  console.log('선택된 값:', e.target.value);
});

With Label & Helper

.select-field로 라벨 + select + 헬퍼 텍스트를 수직으로 묶습니다. .select-label은 for 속성으로 select와 연결하고, aria-describedby로 헬퍼 텍스트를 스크린리더에 전달합니다.

서비스를 제공하는 국가를 선택하세요.

인터페이스 표시 언어를 선택합니다.

알림 및 예약 기능에 사용됩니다. 

<!-- 라벨 + Select + 헬퍼 텍스트 조합 -->
<div class="select-field">
  <label class="select-label" for="country">
    국가
    <span class="select-required" aria-hidden="true">*</span>
  </label>
  <div class="select-wrap">
    <select
      data-select
      id="country"
      aria-required="true"
      aria-describedby="country-helper"
    >
      <option value="">국가를 선택하세요</option>
      <option value="kr" selected>대한민국</option>
      <option value="us">미국</option>
      <option value="jp">일본</option>
    </select>
    <span class="select-icon" aria-hidden="true">
      <i data-lucide="chevron-down"></i>
    </span>
  </div>
  <p class="select-helper" id="country-helper">
    서비스를 제공하는 국가를 선택하세요.
  </p>
</div>
/* select 필드 그룹 */
.select-field {
  display: flex;
  flex-direction: column;
  gap: 4px;
}

/* 라벨 */
.select-label {
  display: inline-flex;
  align-items: center;
  gap: 2px;
  font-size: var(--font-size-sm);
  font-weight: 500;
  color: var(--color-text);
}

/* 필수 표시 */
.select-required {
  color: var(--color-danger);
  margin-left: 1px;
}

/* 헬퍼 텍스트 */
.select-helper {
  font-size: var(--font-size-sm);
  color: var(--color-text-secondary);
  margin: 0;
}
// 국가 선택 변경 시 처리 예시
const countrySelect = document.querySelector('#country');
countrySelect.addEventListener('change', (e) => {
  const selectedCountry = e.target.value;
  console.log('선택된 국가:', selectedCountry);
});

Error & Success State

유효성 검사 실패 시 data-error="true"와 aria-invalid="true"를 함께 적용합니다. 유효성 통과 시 data-success="true"로 성공 상태를 표시합니다. 에러 메시지는 role="alert"으로 스크린리더에 즉시 전달됩니다.

카테고리를 선택해 주세요.

이 서비스는 18세 이상만 이용 가능합니다.

유효한 플랜이 선택되었습니다. 

<!-- 에러 상태 -->
<div class="select-field">
  <label class="select-label" for="category">
    카테고리
    <span class="select-required" aria-hidden="true">*</span>
  </label>
  <div class="select-wrap">
    <select
      data-select data-error="true"
      id="category"
      aria-required="true"
      aria-invalid="true"
      aria-describedby="category-error"
    >
      <option value="">카테고리를 선택하세요</option>
      <option value="tech">기술</option>
    </select>
    <span class="select-icon" aria-hidden="true">
      <i data-lucide="chevron-down"></i>
    </span>
  </div>
  <p class="select-error-msg" id="category-error" role="alert">
    <i data-lucide="alert-circle" aria-hidden="true"></i>
    카테고리를 선택해 주세요.
  </p>
</div>

<!-- 성공 상태 -->
<div class="select-field">
  <label class="select-label" for="plan">구독 플랜</label>
  <div class="select-wrap">
    <select
      data-select data-success="true"
      id="plan"
      aria-describedby="plan-helper"
    >
      <option value="pro" selected>Pro 플랜</option>
    </select>
    <span class="select-icon" aria-hidden="true">
      <i data-lucide="chevron-down"></i>
    </span>
  </div>
  <p class="select-helper select-helper--success" id="plan-helper">
    유효한 플랜이 선택되었습니다.
  </p>
</div>
/* 에러 상태 */
select[data-select][data-error="true"] {
  border-color: var(--color-danger);
  background-color: color-mix(
    in srgb, var(--color-danger) 6%, var(--color-bg)
  );
}

select[data-select][data-error="true"] ~ .select-icon {
  color: var(--color-danger);
}

.select-error-msg {
  display: flex;
  align-items: center;
  gap: 4px;
  font-size: var(--font-size-sm);
  color: var(--color-danger);
  margin: 0;
}

/* 성공 상태 */
select[data-select][data-success="true"] {
  border-color: var(--color-success);
  background-color: color-mix(
    in srgb, var(--color-success) 6%, var(--color-bg)
  );
}

.select-helper--success {
  color: var(--color-success);
}
// 폼 제출 시 select 유효성 검사
const form = document.querySelector('form');
form.addEventListener('submit', (e) => {
  const sel = document.querySelector('[data-select]');
  const errorMsg = document.querySelector('#category-error');

  if (!sel.value) {
    e.preventDefault();
    sel.dataset.error = 'true';
    sel.setAttribute('aria-invalid', 'true');
    errorMsg.hidden = false;
    sel.focus();
  } else {
    delete sel.dataset.error;
    sel.setAttribute('aria-invalid', 'false');
    errorMsg.hidden = true;
  }
});

Disabled State

disabled 속성 추가 시 반투명 처리되고 상호작용이 차단됩니다. 폼 내 편집 불가 항목에 사용하며, 읽기 전용 정보 표시 시에도 활용합니다.

현재 선택할 수 없는 항목입니다.

현재 사용 중인 요금제입니다. 변경하려면 고객센터에 문의하세요. 

<!-- 비활성 select -->
<div class="select-field">
  <label class="select-label" for="region">
    지역 (비활성)
  </label>
  <div class="select-wrap">
    <select
      data-select
      id="region"
      disabled
      aria-disabled="true"
    >
      <option value="">지역을 선택하세요</option>
      <option value="seoul">서울</option>
      <option value="busan">부산</option>
    </select>
    <span class="select-icon" aria-hidden="true">
      <i data-lucide="chevron-down"></i>
    </span>
  </div>
  <p class="select-helper">
    현재 선택할 수 없는 항목입니다.
  </p>
</div>
/* 비활성 상태 */
select[data-select]:disabled {
  opacity: 0.5;
  cursor: not-allowed;
  pointer-events: none;
  background-color: var(--color-bg-secondary);
}
// 조건부 비활성화 예시
const regionSelect = document.querySelector('#region');
const isPlanActive = false;

if (!isPlanActive) {
  regionSelect.disabled = true;
  regionSelect.setAttribute('aria-disabled', 'true');
}

Option Group

optgroup 요소로 관련 옵션들을 시각적으로 그룹핑합니다. label 속성으로 그룹명을 표시하며, 브라우저가 기본으로 지원합니다. 옵션이 많을 때 사용자가 원하는 항목을 더 빠르게 찾을 수 있습니다.

지역별로 도시가 그룹핑되어 있습니다.

용도별로 언어가 분류되어 있습니다.

준비 중인 기능은 선택할 수 없습니다.

<!-- optgroup으로 옵션 그룹핑 -->
<div class="select-field">
  <label class="select-label" for="city">도시</label>
  <div class="select-wrap">
    <select data-select id="city">
      <option value="">도시를 선택하세요</option>
      <optgroup label="수도권">
        <option value="seoul">서울</option>
        <option value="incheon">인천</option>
        <option value="suwon">수원</option>
      </optgroup>
      <optgroup label="영남권">
        <option value="busan">부산</option>
        <option value="daegu">대구</option>
      </optgroup>
    </select>
    <span class="select-icon" aria-hidden="true">
      <i data-lucide="chevron-down"></i>
    </span>
  </div>
</div>

<!-- 비활성 option: 특정 항목만 선택 불가 -->
<optgroup label="고급 기능 (Pro 이상)">
  <option value="ai" disabled>AI 분석 (준비 중)</option>
  <option value="batch">일괄 처리</option>
</optgroup>
/* optgroup 스타일은 브라우저 기본 지원
   select[data-select] 기본 스타일 그대로 적용 */

/* 비활성 option: 브라우저가 자동으로 흐리게 표시
   필요 시 색상 명시 가능 */
select[data-select] option:disabled {
  color: var(--color-text-secondary);
}
// optgroup 동적 필터링 예시
const citySelect = document.querySelector('#city');

function filterByRegion(region) {
  const groups = citySelect.querySelectorAll('optgroup');
  groups.forEach((group) => {
    group.hidden = group.label !== region;
  });
}

// 예: "수도권"만 보이기
filterByRegion('수도권');
